## ğŸ” Sample Mixed-Day Queries

### Find All Mixed Days
```sql
SELECT 
  DATE(created_at) as day,
  COUNT(*) as total_txns,
  COUNT(CASE WHEN transaction_type = 'income' THEN 1 END) as income,
  COUNT(CASE WHEN transaction_type IN ('payment', 'transfer') THEN 1 END) as expenses
FROM transactions
WHERE description LIKE '%HISTORICAL_GEN_2025%'
GROUP BY DATE(created_at)
HAVING COUNT(CASE WHEN transaction_type = 'income' THEN 1 END) > 0
  AND COUNT(CASE WHEN transaction_type IN ('payment', 'transfer') THEN 1 END) > 0
ORDER BY day DESC;
```

### Get Mixed-Day Calendar Data
```sql
WITH daily_summary AS (
  SELECT 
    DATE(created_at) as day,
    SUM(CASE WHEN transaction_type = 'income' THEN amount ELSE 0 END) as total_income,
    SUM(CASE WHEN transaction_type IN ('payment', 'transfer') THEN amount ELSE 0 END) as total_expenses
  FROM transactions
  WHERE description LIKE '%HISTORICAL_GEN_2025%'
  GROUP BY DATE(created_at)
)
SELECT 
  day,
  total_income,
  total_expenses,
  (total_income - total_expenses) as net_change,
  CASE 
    WHEN total_income > 0 AND total_expenses > 0 THEN 'GRADIENT ğŸŸ£'
    WHEN total_income > 0 THEN 'GREEN ğŸŸ¢'
    ELSE 'RED ğŸ”´'
  END as calendar_color
FROM daily_summary
WHERE total_income > 0 AND total_expenses > 0
ORDER BY day DESC;
```
